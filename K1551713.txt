; NAME: SHOW
; INSTRUCTION: Print a string using function 9.
; REG OCCUPATION: AX, DX.
; INPUT: Already initialized string.
; OUTPUT: None.
; EXAMPLE: SHOW STR1

SHOW MACRO STR
	PUSH	AX
	PUSH	DX
	LEA		DX, STR
	MOV		AH, 9
	INT		21H
	POP		DX
	POP		AX
ENDM
